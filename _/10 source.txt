    var suits = ["spades", "hearts", "clubs", "diams"];

      if(firstRun){
        firstRun = false;
        buildCards();
        shuffleArray(cards);
        dealCards(cards);
      }
      console.log('works');
    }

    function buildCards() {
      cards = [];
      for(s in suits){
        var suitNew = suits[s][0].toUpperCase();
        for(n in cardFace){
          var card = {
            suit:suits[s],
            num:cardFace[n],
            cardValue:parseInt(n) +2,
            icon:suitNew
          }
          cards.push(card);
        }
      }
      console.log(cards);
    }

    function dealCards(array){
      for(var i =0;i<array.length;i++){
        var m = i % 2;
        players[m].push(array[i]);
      }
    }

    function shuffleArray(array){
       for(var x = array.length -1;x>0;x--){
         var ii = Math.floor(Math.random() * (x+1));
         var temp = array[x];
         array[x] = array[ii];
         array[ii] = temp;
       }
       return array;
    }
